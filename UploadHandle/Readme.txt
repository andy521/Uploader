#上传处理前后台对接信息说明
1.前台发送文件信息
{
 oldName:'', //上传的文件名称
 newName:'',//后台生成的文件名称
 size:'',//上传文件的大小
 handleType:'',//当前上传的后台处理模式  1. single 简单处理模式 2.temporary 带临时处理
 subfolder:'',//指定后台保存的字文件夹
 other:'',//上传的其他参数或说明
}
2.后台相应接收信息
{
 status:'', //相应状态 1--成功 2--有错误
 newName:'',// 当前上传的文件名
 relativeName:'',//当前上传的文件相对路径
 curSize:'', //当前已经保存的数据大小
 curLength:'',当前发送数据大小
 msg:'',// 上传成功或失败的描述
}
